#include <iostream>
using namespace std;
// Программа для создания и вывода таблицы умножения 10x10
int main() {
  int array[10][10];
  for (int i = 0; i < 10; i++) {
    for (int j = 0; j < 10; j++) {
      array[i][j] = (i + 1) * (j + 1);
    }
  }
  // вывод таблицу умножения
  cout << "Таблица умножения 10x10:" << endl;
  for (int i = 0; i < 10; i++) {
    for (int j = 0; j < 10; j++) {
      cout << array[i][j] << "\t";
    }
    cout << endl;
  }
  // ввывод задания пользователю
  cout << "\nПоробуй решить примеры!" << endl;
  cout << "Формат ввода 2 2 4" << endl;
  cout << "Для ввода веди 0 0 0" << endl;
  // цикл для решения примеров
  while (true) {
    int a, b, otvet;
    cout << "Введите пример: ";
    cin >> a >> b >> otvet;

    if (a == 0 && b == 0 && otvet == 0) {
      cout << "Выход из программы, да ты молодец тоесть я))" << endl;
      break;
    }
    // проверка ответа пользователя
    int pravilni_otvet = a * b;
    if (otvet == pravilni_otvet) {
      cout << "Правильно!" << a << " * " << b << " = " << otvet << endl;
    } else {
      cout << "Неправильно! " << a << " * " << b << " = " << pravilni_otvet
           << ", а не " << otvet << endl;
    }
  }

  return 0;
}
